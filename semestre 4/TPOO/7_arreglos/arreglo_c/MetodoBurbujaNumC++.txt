/*metodo de la burbuja arreglo unidimensional*/

#include<stdio.h>

#define max 10

void main(){

				 unsigned short int i, cambio, n;

				 int aux, a[max];

				 do{

					 puts("ingrese tamaño de arreglo unidimensional:");

					 scanf("%d", &n);

					 }while(n>max&&n<2);

				 for(i=0;i<n;i++){

										printf("a[%d]=", i);

										scanf("%d", &a[i]);

										}

				 cambio=1;

				 while(cambio==1){

										cambio=0;

										for(i=0;i<n-1;i++){

																if(a[i]>a[i+1]){

																					 aux=a[i];

																					 a[i]=a[i+1];

																					 a[i+1]=aux;

																					 cambio=1;

																					 }

																}

										}

				 puts("el arreglo quedo asi:");

				 for(i=0;i<n;i++)

									  printf("%d, ", a[i]);
             }
